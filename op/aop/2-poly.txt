class Publication{
    protected String title;
    protected double price;
    protected int copies;
    Publication(String title,double price,int copies){
        this.title=title;
        this.price=price;
        this.copies=copies;
    }
    public void saleCopy(){
        system.out.println("Total sell :"+price*copies);
    }
    public void displayinfo(){
        system.out.println("Title : "+title+
                        "Price : "+price+      
                        "Copies : "+copies      
                             );}
}
class Book extends Publication{
    protected String author;
    Book(String title,double price,int copies,String author){
        super(title,price,copies);
        this.author=author;
    }
    void orderCopies(int count){
        this.copies=copies+count;
        System.out.println("Total order Copies : "+copies);
    }
    
}
